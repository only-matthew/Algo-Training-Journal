#include<cstdio>
#include<iostream>
long long n,m,z,jvv;
int main(){
    scanf("%d %d",&n,&m);
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            if(i==j)
                z+=(n-i)*(m-j);
            else
                jvv+=(n-i)*(m-j);
        }
    }
//    printf("%ld %ld",z,jvv);
    std::cout<<z<<" "<<jvv;
}
