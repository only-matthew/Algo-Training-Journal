#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
struct Point{
    int x, y ,z;
    const bool operator < (const Point &other){
        return z < other.z;
    }
};
const int MAXN = 5e4 + 5;
Point a[MAXN];
int main(){
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> a[i].x >> a[i].y >> a[i].z;
    sort (a + 1, a + n + 1);
    double res = 0;
    for (int i = 2; i <= n; i++){
        res += sqrt(pow(a[i].x - a[i-1].x,2) + pow(a[i].y - a[i - 1].y,2) + pow(a[i].z - a[i - 1].z, 2)); 
    }
    printf("%.3f", res);
    return 0;
}
