#include<algorithm>

using namespace std;
struct Person{
    string s;
    int num, y, m, d;
    const bool operator <(const Person &other){
        if (y != other.y) return y < other.y;
        else if (m != other.m) return m < other.m;
        else if (d != other.d) return d < other.d;
        else return num > other.num;
    }
};
const int MAXN = 100 + 5;
Person a[MAXN];
int main(){
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> a[i].s >> a[i].y >> a[i].m >> a[i].d;
        a[i].num = i;
    }
    sort(a + 1, a + n + 1);
    for (int i = 1; i <= n; i++){
        cout << a[i].s << endl;
    }
    return 0;
}
